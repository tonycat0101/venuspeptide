function enviarPedidoFormulario(e) { e.preventDefault(); // ── VALIDACIÓN DE CARRITO (LÍMITES) ── const cart = JSON.parse(localStorage.getItem('venus_cart')) || []; // Validar máximo 30 unidades totales const totalItems = cart.reduce((sum, item) => sum + item.qty, 0); if (totalItems > 30) { alert(`⚠️ Máximo 30 unidades por orden. Actualmente tienes ${totalItems} unidades.`); return; } // Validar máximo 10 por producto for (const item of cart) { if (item.qty > 10) { const prod = window.productosData[item.id]; alert(`⚠️ Máximo 10 unidades de ${prod.nombre}. Tienes ${item.qty} en el carrito.`); return; } } // ── VALIDACIONES DEL FORMULARIO ── const legalChecked = document.getElementById('checkoutLegal').checked; if (!legalChecked) { document.getElementById('legalError').classList.add('visible'); return; } const nombre = document.getElementById('checkoutNombre').value.trim(); if (!nombre || nombre.length < 2) { alert('Por favor, ingresa tu nombre completo.'); return; } const email = document.getElementById('checkoutEmail').value.trim(); if (!email || !email.includes('@') || !email.includes('.')) { alert('Por favor, ingresa un correo electrónico válido.'); return; } const telefono = document.getElementById('checkoutTelefono').value.trim(); if (!telefono || telefono.length < 7) { alert('Por favor, ingresa un número de teléfono válido.'); return; } const direccion = document.getElementById('checkoutDireccion').value.trim(); if (!direccion || direccion.length < 5) { alert('Por favor, ingresa una dirección completa.'); return; } const ciudad = document.getElementById('checkoutCiudad').value.trim(); if (!ciudad || ciudad.length < 2) { alert('Por favor, ingresa tu ciudad.'); return; } const estado = document.getElementById('checkoutEstado').value.trim(); if (!estado || estado.length < 2) { alert('Por favor, ingresa tu estado o provincia.'); return; } const zip = document.getElementById('checkoutZip').value.trim(); if (!zip || zip.length < 3) { alert('Por favor, ingresa un código postal válido.'); return; } // ── PROCESAR PEDIDO ── const metodoPago = document.querySelector('input[name="metodo_pago"]:checked').value; const notas = document.getElementById('checkoutNotas').value.trim(); const newsletter = document.getElementById('checkoutNewsletter').checked; let subtotal = 0; cart.forEach(item => { const prod = window.productosData ? window.productosData[item.id] : null; if (prod) { subtotal += prod.precio * item.qty; } }); const totalConDescuento = subtotal - descuentoAplicado; const tax = totalConDescuento * 0.06; const totalFinal = totalConDescuento + tax; const codigoReferencia = 'VP-' + Math.floor(1000 + Math.random() * 9000); // ── GUARDAR EN SESSIONSTORAGE ── sessionStorage.setItem('pending_order_id', codigoReferencia); sessionStorage.setItem('pending_order_total', totalFinal.toFixed(2)); sessionStorage.setItem('pending_order_method', metodoPago); // ── BACKUP EN LOCALSTORAGE ── localStorage.setItem('backup_order_id', codigoReferencia); localStorage.setItem('backup_order_total', totalFinal.toFixed(2)); localStorage.setItem('backup_order_method', metodoPago); // ── GUARDAR DATOS DEL CLIENTE ── localStorage.setItem('checkout_nombre', nombre); localStorage.setItem('checkout_email', email); localStorage.setItem('checkout_telefono', telefono); localStorage.setItem('checkout_direccion', direccion); localStorage.setItem('checkout_ciudad', ciudad); localStorage.setItem('checkout_estado', estado); localStorage.setItem('checkout_zip', zip); localStorage.setItem('checkout_notas', notas); localStorage.setItem('checkout_newsletter', newsletter ? 'si' : 'no'); // ── GUARDAR EN SESSIONSTORAGE (para instrucciones) ── sessionStorage.setItem('checkout_nombre', nombre); sessionStorage.setItem('checkout_email', email); sessionStorage.setItem('checkout_telefono', telefono); // ── GUARDAR EN FIREBASE ── const db = window.db; const doc = window.doc; const setDoc = window.setDoc; if (db && doc && setDoc) { const orderRef = doc(db, 'pedidos', codigoReferencia); setDoc(orderRef, { codigo: codigoReferencia, cliente: nombre, email: email, telefono: telefono, direccion: direccion, ciudad: ciudad, estado: estado, zip: zip, productos: cart, subtotal: subtotal, descuento: descuentoAplicado, tax: tax, total: totalFinal, metodo: metodoPago, notas: notas, newsletter: newsletter, codigo_promo: codigoPromoUsado, status: 'pendiente', fecha: new Date().toISOString() }).then(() => { console.log('✅ Orden guardada en Firebase correctamente'); window.location.href = 'instrucciones.html'; }).catch((error) => { console.error('❌ Error al guardar en Firebase:', error); alert('Hubo un error al procesar tu pedido. Por favor, intenta de nuevo.'); }); } else { // Fallback si Firebase no está disponible console.warn('⚠️ Firebase no disponible, redirigiendo sin guardar'); window.location.href = 'instrucciones.html'; } }
